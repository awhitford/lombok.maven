3   import java.io.File;
4   import java.io.IOException;
5   import java.lang.reflect.InvocationTargetException;
6   import java.nio.charset.Charset;
7   import java.nio.charset.UnsupportedCharsetException;
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Map;
11  
12  import org.apache.commons.lang3.StringUtils;
13  import org.apache.maven.artifact.Artifact;
14  import org.apache.maven.plugin.AbstractMojo;
15  import org.apache.maven.plugin.MojoExecutionException;
16  import org.apache.maven.plugin.logging.Log;
17  import org.apache.maven.plugins.annotations.Parameter;
18  import org.apache.maven.project.MavenProject;
19  
20  import lombok.launch.Delombok;
21  
22  /**
23   * Abstract mojo to Delombok java source with lombok annotations.
24   *
25   * @author <a href="mailto:anthony@whitford.com">Anthony Whitford</a>
26   * @see <a href="http://projectlombok.org/features/delombok.html">Delombok</a>
27   */
28  public abstract class AbstractDelombokMojo extends AbstractMojo {
29  
30      /**
31       * Specifies whether the delombok generation should be skipped.
32       */
33      @Parameter(property="lombok.delombok.skip", defaultValue="false", required=true)
34      protected boolean skip;
35  
36      /**
37       * Encoding.
38       */
39      @Parameter(property="lombok.encoding", defaultValue="${project.build.sourceEncoding}", required=true)
40      protected String encoding;
41  
42      /**
43       * Verbose flag.  Print the name of each file as it is being delombok-ed.
44       */
45      @Parameter(property="lombok.verbose", defaultValue="false", required=true)
46      protected boolean verbose;
47   
48      /**
49       * Add output directory flag.  Adds the output directory to the Maven build path.
50       */
51      @Parameter(property="lombok.addOutputDirectory", defaultValue="true", required=true)
52      protected boolean addOutputDirectory;
53  
54      /**
55       * Formatting preferences.
56       */
57      @Parameter
58      protected Map<String, String> formatPreferences;
59  
60      /**
61       * The Maven project to act upon.
62       */
63      @Parameter(property="project", required=true, readonly=true)
64      protected MavenProject project;
65  
66      /**
67       * The plugin dependencies.
68       */
69      @Parameter(property="plugin.artifacts", required=true, readonly=true)
70      private List<Artifact> pluginArtifacts;
71  
72      protected abstract String getGoalDescription ();
73  
74      protected abstract File getOutputDirectory();
75  
76      protected abstract File getSourceDirectory();
77  
78      protected abstract String getSourcePath();
79  
80      protected abstract void addSourceRoot(String path);
81  
82      @Override
83      public void execute() throws MojoExecutionException {
84          final Log logger = getLog();
85          assert null != logger;
86  
87          final String goal = getGoalDescription();
88          logger.debug("Starting " + goal);
89          final File outputDirectory = getOutputDirectory();
90          logger.debug("outputDirectory: " + outputDirectory);
91          final File sourceDirectory = getSourceDirectory();
92          logger.debug("sourceDirectory: " + sourceDirectory);
93          final String sourcePath = getSourcePath();
94          logger.debug("sourcePath: " + sourcePath);
95  
96          if (this.skip) {
97              logger.warn("Skipping " + goal);
98          } else if (sourceDirectory.exists()) {
99              // Build a classPath for delombok...
100             final StringBuilder classPathBuilder = new StringBuilder();
101             for (final Object artifact : project.getArtifacts()) {
102                 classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
103             }
104             for (final Artifact artifact : pluginArtifacts) {
105                 classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
106             }
107             final String classPath = classPathBuilder.toString();
108             logger.debug("classpath: " + classPath);
109             try {
110                 final Delombok delombok = new Delombok();
111                 delombok.setVerbose(this.verbose);
112                 delombok.setClasspath(classPath);
113 
114                 if (StringUtils.isNotBlank(this.encoding)) {
115                     try {
116                         delombok.setCharset(this.encoding);
117                     } catch (final UnsupportedCharsetException e) {
118                         logger.error("The encoding parameter is invalid; Please check!", e);
119                         throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
120                     }
121                 } else {
122                     logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
123                 }
124 
125                 if (null != formatPreferences && !formatPreferences.isEmpty()) {
126                     try {
127                         // Construct a list array just like the command-line option...
128                         final List<String> formatOptions = new ArrayList<String>(formatPreferences.size());
129                         for (final Map.Entry<String, String> entry : formatPreferences.entrySet()) {
130                             final String key = entry.getKey();
131                             // "pretty" is an exception -- it has no value...
132                             formatOptions.add( "pretty".equalsIgnoreCase(key) ? key : (key + ':' + entry.getValue()) );
133                         }
134                         delombok.setFormatPreferences(delombok.formatOptionsToMap(formatOptions));
135                     } catch (final Exception e) {
136                         logger.error("The formatPreferences parameter is invalid; Please check!", e);
137                         throw new MojoExecutionException("Invalid formatPreferences: " + this.formatPreferences, e);
138                     }
139                 }
140 
141                 try {
142                     delombok.setOutput(outputDirectory);
143                     delombok.setSourcepath(getSourcePath());
144                     delombok.addDirectory(sourceDirectory);
145                     delombok.delombok();
146                     logger.info(goal + " complete.");
147                     
148                     if (this.addOutputDirectory) {
149                         // adding generated sources to Maven project
150                         addSourceRoot(outputDirectory.getCanonicalPath());
151                     }
152                 } catch (final IOException e) {
153                     logger.error("Unable to delombok!", e);
154                     throw new MojoExecutionException("I/O problem during delombok", e);
155                 }
156             } catch (final ClassNotFoundException e) {
157                 throw new MojoExecutionException("Unable to delombok", e);
158             } catch (final IllegalAccessException e) {
159                 throw new MojoExecutionException("Unable to delombok", e);
160             } catch (final InvocationTargetException e) {
161                 throw new MojoExecutionException("Unable to delombok", e);
162             } catch (final InstantiationException e) {
163                 throw new MojoExecutionException("Unable to delombok", e);
164             } catch (final NoSuchMethodException e) {
165                 throw new MojoExecutionException("Unable to delombok", e);
166             }
167         } else {
168             logger.warn("Skipping " + goal + "; no source to process.");
169         }
170     }
171 }
