001    package lombok.maven;
002    
003    import java.io.File;
004    import java.io.IOException;
005    import java.nio.charset.Charset;
006    import java.nio.charset.UnsupportedCharsetException;
007    import java.util.List;
008    
009    import org.apache.commons.lang3.StringUtils;
010    import org.apache.maven.artifact.Artifact;
011    import org.apache.maven.plugin.AbstractMojo;
012    import org.apache.maven.plugin.MojoExecutionException;
013    import org.apache.maven.plugin.logging.Log;
014    import org.apache.maven.project.MavenProject;
015    
016    import lombok.delombok.Delombok;
017    
018    
019    /**
020     * Abstract mojo to Delombok java source with lombok annotations.
021     *
022     * @threadSafe
023     * @author <a href="mailto:anthony@whitford.com">Anthony Whitford</a>
024     * @see <a href="http://projectlombok.org/features/delombok.html">Delombok</a>
025     */
026    public abstract class AbstractDelombokMojo extends AbstractMojo {
027    
028        /**
029         * Specifies whether the delombok generation should be skipped.
030         * @parameter expression="${lombok.delombok.skip}" default-value="false"
031         * @required
032         */
033        protected boolean skip;
034    
035        /**
036         * Encoding.
037         * @parameter expression="${lombok.encoding}" default-value="${project.build.sourceEncoding}"
038         * @required
039         */
040        protected String encoding;
041    
042        /**
043         * Verbose flag.  Print the name of each file as it is being delombok-ed.
044         * @parameter expression="${lombok.verbose}" default-value="false"
045         * @required
046         */
047        protected boolean verbose;
048    
049        /**
050         * The Maven project to act upon.
051         * @parameter expression="${project}"
052         * @required
053         * @readonly
054         */
055        protected MavenProject project;
056    
057        /**
058         * The plugin dependencies.
059         * 
060         * @parameter expression="${plugin.artifacts}"
061         * @required
062         * @readonly
063         */
064        protected List<Artifact> pluginArtifacts;
065    
066        protected abstract String getGoalDescription ();
067    
068        protected abstract File getOutputDirectory();
069    
070        protected abstract File getSourceDirectory();
071    
072        protected abstract String getSourcePath();
073    
074        protected abstract void addSourceRoot(String path);
075    
076        @Override
077        public void execute() throws MojoExecutionException {
078            final Log logger = getLog();
079            assert null != logger;
080    
081            final String goal = getGoalDescription();
082            logger.debug("Starting " + goal);
083            final File outputDirectory = getOutputDirectory();
084            logger.debug("outputDirectory: " + outputDirectory);
085            final File sourceDirectory = getSourceDirectory();
086            logger.debug("sourceDirectory: " + sourceDirectory);
087            final String sourcePath = getSourcePath();
088            logger.debug("sourcePath: " + sourcePath);
089    
090            if (this.skip) {
091                logger.warn("Skipping " + goal);
092            } else if (sourceDirectory.exists()) {
093                // Build a classPath for delombok...
094                final StringBuilder classPathBuilder = new StringBuilder();
095                for (final Object artifact : project.getArtifacts()) {
096                    classPathBuilder.append(((Artifact)artifact).getFile()).append(File.pathSeparatorChar);
097                }
098                for (final Artifact artifact : pluginArtifacts) {
099                    classPathBuilder.append(artifact.getFile()).append(File.pathSeparatorChar);
100                }
101                final String classPath = classPathBuilder.toString();
102                logger.debug("classpath: " + classPath);
103                final Delombok delombok = new Delombok();
104                delombok.setVerbose(this.verbose);
105                delombok.setClasspath(classPath);
106    
107                if (StringUtils.isNotBlank(this.encoding)) {
108                    try {
109                        delombok.setCharset(this.encoding);
110                    } catch (final UnsupportedCharsetException e) {
111                        logger.error("The encoding parameter is invalid; Please check!", e);
112                        throw new MojoExecutionException("Unknown charset: " + this.encoding, e);
113                    }
114                } else {
115                    logger.warn("No encoding specified; using default: " + Charset.defaultCharset());
116                }
117    
118                try {
119                    delombok.setOutput(outputDirectory);
120                    delombok.setSourcepath(getSourcePath());
121                    delombok.addDirectory(sourceDirectory);
122                    delombok.delombok();
123                    logger.info(goal + " complete.");
124    
125                    // adding generated sources to Maven project
126                    addSourceRoot(outputDirectory.getCanonicalPath());
127                } catch (final IOException e) {
128                    logger.error("Unable to delombok!", e);
129                    throw new MojoExecutionException("I/O problem during delombok", e);
130                }
131            } else {
132                logger.warn("Skipping " + goal + "; no source to process.");
133            }
134        }
135    }
